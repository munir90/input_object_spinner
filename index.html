Input Object / Text Values Spinner

Initializing Spinner 

$(input).InputObjectSpinner("init", {
    data: [
        {
            "text": "Option 1",
            "value": "op_val_1",
            "color_class": "bg-success",
            "description": "Option 1 Description",
            "default": true
        },
        {
            "text": "Option 2",
            "value": "op_val_2",
            "color_class": "bg-warning",
            "description": "Option 2 Description"
        }
        {
            "text": "Option 3",
            "value": "op_val_3",
            "color_class": "bg-danger",
            "description": "Option 3 Description"
        }
    ]
});

Getting Selected Data

$(input).InputObjectSpinner("getSelectedData");

Set Value?

$(input).setInputObjectValue("op_val_2");

Check if input is object spinner?

let isInputObjectSpinner = $(input).InputObjectSpinner("isInputObjectSpinner");

if (isInputObjectSpinner === true) {
    console.log("This input is initialized as object spinner.");
} else {
    console.log("input is not an object spinner.");
}
